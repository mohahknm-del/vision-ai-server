import express from "express"; import fetch from "node-fetch"; import cors from "cors"; const app = express(); app.use(cors()); app.use(express.json()); // 🔑 ضع مفتاحك الحقيقي من حسابك Pro هنا const HF_TOKEN = process.env.HF_TOKEN app.post("/generate", async (req, res) => { try { const { prompt } = req.body; const response = await fetch( "https://api-inference.huggingface.co/models/SG161222/Realistic_Vision_V5.1", { method: "POST", headers: { Authorization: `Bearer ${HF_TOKEN}`, "Content-Type": "application/json", }, body: JSON.stringify({ inputs: prompt }), } ); if (!response.ok) { const err = await response.text(); console.error("HF Error:", err); return res.status(400).json({ error: "خطأ من واجهة Hugging Face", details: err }); } const buffer = await response.arrayBuffer(); res.setHeader("Content-Type", "image/png"); res.send(Buffer.from(buffer)); } catch (e) { console.error("Server error:", e); res.status(500).json({ error: "فشل الاتصال بواجهة Hugging Face" }); } }); app.listen(3000, () => console.log("✅ AI server ready on port 3000 (using Realistic Vision V5.1 model)"));
